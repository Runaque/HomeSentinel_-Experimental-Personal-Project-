# HomeSentinel — main component
# Registers all firmware modules and embeds the gzipped web UI into flash.
#
# The web/ assets are gzipped and embedded as binary blobs so the HTTP server
# can serve them straight from flash with Content-Encoding: gzip — no
# filesystem read on the hot path.
#
# REQUIRES vs PRIV_REQUIRES: none of this component's *public* headers
# (inventory.h, discovery.h, etc.) pull in other components' headers — the
# dependencies are all used inside our .c files. So every dependency belongs
# in PRIV_REQUIRES (private). This keeps the component's public interface clean.

idf_component_register(
    SRCS
        "main.c"
        "wifi.c"
        "discovery.c"
        "inventory.c"
        "anomaly.c"
        "oui.c"
        "web.c"
        "notifier.c"
    INCLUDE_DIRS
        "."
    PRIV_REQUIRES
        esp_wifi
        esp_netif
        lwip
        mdns
        esp_http_server
        esp_http_client
        esp_timer
        esp-tls
        esp_partition
        mbedtls
        nvs_flash
        json
    EMBED_FILES
        "${CMAKE_CURRENT_SOURCE_DIR}/../web/index.html.gz"
)
